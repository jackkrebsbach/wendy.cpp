cmake_minimum_required(VERSION 3.15)
project(wendy LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

file(GLOB WENDY_SOURCES src/*.cpp)
list(APPEND WENDY_SOURCES example.cpp)

find_package(spdlog REQUIRED)
find_package(fmt REQUIRED)
find_package(xtensor REQUIRED)

add_executable(example example.cpp ${WENDY_SOURCES})

target_include_directories(example PRIVATE /opt/homebrew/include)
target_link_directories(example PRIVATE /opt/homebrew/lib)

target_link_libraries(example
        PRIVATE
        symengine
        flint
        mpfr
        gmp
        spdlog::spdlog
        fmt::fmt
        xtensor
)
