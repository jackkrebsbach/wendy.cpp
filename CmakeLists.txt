cmake_minimum_required(VERSION 3.15)
project(wendy LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

file(GLOB_RECURSE WENDY_SOURCES src/*.cpp)
list(APPEND WENDY_SOURCES example.cpp)

add_executable(example ${WENDY_SOURCES})

find_library(FLINT_LIBRARY flint REQUIRED)
find_library(MPFR_LIBRARY mpfr REQUIRED)
find_library(GMP_LIBRARY gmp REQUIRED)
find_library(FFTW3_LIBRARY fftw3 REQUIRED)

find_package(BLAS REQUIRED)
find_package(LAPACK REQUIRED)
find_package(xtensor REQUIRED)
find_package(symengine REQUIRED)
find_package(xtensor-blas REQUIRED)
find_package(OpenMP REQUIRED)
find_package(Eigen3 REQUIRED)

include_directories(external/CppNumericalSolvers/include)
include_directories(external/exprtk)

target_link_libraries(example
        PRIVATE
        symengine
        xtensor
        xtensor-blas
        Eigen3::Eigen
        ${BLAS_LIBRARIES}
        ${LAPACK_LIBRARIES}
        ${FLINT_LIBRARY}
        ${MPFR_LIBRARY}
        ${GMP_LIBRARY}
        ${FFTW3_LIBRARY}
)
